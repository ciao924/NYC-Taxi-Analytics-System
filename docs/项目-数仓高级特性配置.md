## 一、SCD Type2 实现

### 1.1 SCD Type2 概述

SCD（Slowly Changing Dimensions，缓慢变化维度）Type2 是一种维度历史变化追踪方式，通过添加 `effective_date`（生效日期）、`expire_date`（过期日期）和 `is_current`（是否当前版本）字段来保留维度表的历史版本。

### 1.2 维度表 SCD Type2 配置

**适用表**：
- `dim_location_zone` - 区域维度
- `dim_vendor` - 供应商维度

### 1.3 DDL 变更

```sql
-- 为维度表增加 SCD Type2 字段
ALTER TABLE nyc_taxi_dwd.dim_location_zone
ADD COLUMNS (
  effective_date STRING COMMENT '生效日期，格式: yyyy-MM-dd',
  expire_date STRING COMMENT '过期日期，格式: yyyy-MM-dd，NULL表示永久有效',
  is_current STRING COMMENT '是否当前版本: Y/N'
);

ALTER TABLE nyc_taxi_dwd.dim_vendor
ADD COLUMNS (
  effective_date STRING COMMENT '生效日期',
  expire_date STRING COMMENT '过期日期',
  is_current STRING COMMENT '是否当前版本: Y/N'
);
```

### 1.4 SCD Type2 实现逻辑

```scala
// ScdType2Processor.scala

package com.taxi.etl.common

import org.apache.spark.sql.{DataFrame, SparkSession}
import org.apache.spark.sql.functions._
import org.apache.spark.sql.types._

object ScdType2Processor {

  /**
   * SCD Type2 合并
   * @param spark SparkSession
   * @param dimTableName 维度表名
   * @param newData 新数据
   * @param businessKey 业务主键
   * @param currentDate 当前日期
   */
  def merge(
             spark: SparkSession,
             dimTableName: String,
             newData: DataFrame,
             businessKey: String,
             currentDate: String
           ): Unit = {
    import spark.implicits._

    // 读取现有维度数据
    val existingData = spark.table(dimTableName)

    // 找出需要更新的记录（主键相同但属性变化）
    val changedRecords = newData.join(
      existingData.filter(col("is_current") === "Y"),
      newData(businessKey) === existingData(businessKey)
    ).where(
      // 检查是否有属性变化（排除生效日期）
      newData.columns
        .filter(_ != businessKey)
        .map(c => newData(c) =!= existingData(c))
        .reduce(_ || _)
    )

    // 过期旧记录
    if (changedRecords.count() > 0) {
      val expiredRecords = changedRecords.select(
        existingData.columns.map(c => existingData(c)) : _*
      ).withColumn("expire_date", lit(currentDate))
        .withColumn("is_current", lit("N"))

      // 生成新记录
      val newRecords = newData.join(
        existingData.filter(col("is_current") === "Y")
          .select(businessKey),
        newData(businessKey) === existingData.filter(col("is_current") === "Y")(businessKey)
      ).withColumn("effective_date", lit(currentDate))
        .withColumn("expire_date", lit(null))
        .withColumn("is_current", lit("Y"))

      // 合并数据
      val mergedData = existingData
        .filter(col("is_current") === "N" || !existingData(businessKey).isin(
          changedRecords.select(existingData(businessKey)).as[String].collect(): _*
        ))
        .unionByName(expiredRecords)
        .unionByName(newRecords)

      // 写入维度表
      mergedData.write
        .format("iceberg")
        .mode("overwrite")
        .save(dimTableName)
    } else {
      // 无变化，仅添加新记录
      val insertRecords = newData
        .withColumn("effective_date", lit(currentDate))
        .withColumn("expire_date", lit(null))
        .withColumn("is_current", lit("Y"))

      existingData
        .unionByName(insertRecords)
        .write
        .format("iceberg")
        .mode("overwrite")
        .save(dimTableName)
    }
  }

  /**
   * 查询当前有效记录
   */
  def getCurrentRecords(spark: SparkSession, dimTableName: String): DataFrame = {
    spark.table(dimTableName).filter(col("is_current") === "Y")
  }
}
```

### 2.2 增量 ETL 读取逻辑

```scala
// IncrementalEtlLoader.scala

package com.taxi.etl.common

import org.apache.spark.sql.{DataFrame, SparkSession}
import org.apache.spark.sql.functions._

object IncrementalEtlLoader {

  def readIncrementalData(
                           spark: SparkSession,
                           tableName: String,
                           config: IncrementalEtlConfig,
                           startDate: String,
                           endDate: String
                         ): DataFrame = {
    import spark.implicits._

    val lookbackDate = calculateLookbackDate(startDate, config.lookbackDays)

    val query = if (config.enabled) {
      s"""
        SELECT *
        FROM ${tableName}
        WHERE ${config.partitionColumn} >= '${lookbackDate}'
          AND ${config.partitionColumn} <= '${endDate}'
      """
    } else {
      s"""
        SELECT *
        FROM ${tableName}
        WHERE ${config.partitionColumn} >= '${startDate}'
          AND ${config.partitionColumn} <= '${endDate}'
      """
    }

    spark.sql(query)
  }

  private def calculateLookbackDate(baseDate: String, lookbackDays: Int): String = {
    val date = java.time.LocalDate.parse(baseDate)
    date.minusDays(lookbackDays).toString
  }

  def getPartitionList(
                        spark: SparkSession,
                        tableName: String,
                        startDate: String,
                        endDate: String
                      ): Seq[String] = {
    import spark.implicits._

    val partitions = spark.sql(s"""
      SHOW PARTITIONS ${tableName}
      WHERE dt >= '${startDate}' AND dt <= '${endDate}'
    """).as[String].collect()

    partitions.map(p => p.split("/").last.split("=").last)
  }
}
```

### 3.2 分区配置

```scala
// PartitionConfig.scala

package com.taxi.etl.common

object PartitionConfig {

  case class PartitionSpec(
                            column: String,
                            format: String,
                           二级Partition: Option[String] = None
                          )

  val ODS_PARTITION = PartitionSpec(
    column = "dt",
    format = "yyyy-MM-dd",
    二级分区 = Some("taxi_type")
  )

  val DWD_PARTITION = PartitionSpec(
    column = "dt",
    format = "yyyy-MM-dd"
  )

  val DWS_PARTITION = PartitionSpec(
    column = "dt",
    format = "yyyy-MM-dd"
  )

  def getPartitionSpec(layer: String): PartitionSpec = layer match {
    case "ODS" => ODS_PARTITION
    case "DWD" => DWD_PARTITION
    case "DWS" => DWS_PARTITION
    case _ => DWD_PARTITION
  }

  def formatPartitionValue(value: String, format: String): String = format match {
    case "yyyy-MM-dd" => s"dt=${value}"
    case "yyyy-MM" => s"month=${value}"
    case "yyyy" => s"year=${value}"
    case _ => s"dt=${value}"
  }
}
```

## 五、小文件自动合并任务

### 5.1 小文件合并工具类

```scala
// SmallFileMergeTask.scala

package com.taxi.etl.common

import org.apache.spark.sql.{DataFrame, SparkSession}
import org.slf4j.LoggerFactory

object SmallFileMergeTask {

  private val logger = LoggerFactory.getLogger(getClass)

  case class MergeConfig(
                          tableName: String,
                          tablePath: String,
                          targetFileSizeMB: Long = 256,
                          partitionColumn: Option[String] = Some("dt")
                        )

  val SUPPORTED_TABLES = Seq(
    MergeConfig(
      tableName = "ods_green",
      tablePath = "/user/hive_local/warehouse/nyc_taxi_ods.db/taxi_trip_green_ods",
      partitionColumn = Some("dt")
    ),
    MergeConfig(
      tableName = "ods_yellow",
      tablePath = "/user/hive_local/warehouse/nyc_taxi_ods.db/taxi_trip_yellow_ods",
      partitionColumn = Some("dt")
    ),
    MergeConfig(
      tableName = "dwd_detail",
      tablePath = "/user/hive_local/warehouse/nyc_taxi_dwd.db/dwd_taxi_trip_detail",
      partitionColumn = Some("dt")
    ),
    MergeConfig(
      tableName = "dws_daily",
      tablePath = "/user/hive_local/warehouse/nyc_taxi_dws.db/dws_taxi_daily_stats",
      partitionColumn = Some("dt")
    )
  )

  def main(args: Array[String]): Unit = {
    val spark = SparkSession.builder()
      .enableHiveSupport()
      .config("spark.sql.adaptive.enabled", "true")
      .config("spark.sql.adaptive.coalescePartitions.enabled", "true")
      .getOrCreate()

    try {
      val targetTable = args.lift(0).getOrElse("all")
      val targetDate = args.lift(1).getOrElse(null)

      logger.info(s"开始小文件合并任务: table=$targetTable, date=$targetDate")

      val tablesToProcess = if (targetTable == "all") {
        SUPPORTED_TABLES
      } else {
        SUPPORTED_TABLES.filter(_.tableName == targetTable)
      }

      tablesToProcess.foreach { config =>
        try {
          mergeSmallFiles(spark, config, targetDate)
        } catch {
          case e: Exception =>
            logger.error(s"合并 ${config.tableName} 失败: ${e.getMessage}", e)
        }
      }

      logger.info("小文件合并任务完成")

    } finally {
      spark.stop()
    }
  }

  def mergeSmallFiles(
                       spark: SparkSession,
                       config: MergeConfig,
                       targetDate: String
                     ): Unit = {
    import spark.implicits._

    logger.info(s"处理表: ${config.tableName}")

    // 读取数据
    val basePath = config.tablePath
    val df = if (targetDate != null && config.partitionColumn.isDefined) {
      spark.read
        .format("parquet")
        .load(s"${basePath}/${config.partitionColumn.get}=${targetDate}")
    } else {
      spark.read.format("parquet").load(basePath)
    }

    val totalSizeMB = df.rdd.getNumPartitions * 128.0 / 1024
    logger.info(s"当前数据: ${df.count()} 条, ${df.rdd.getNumPartitions} 分区, 估算大小 ${totalSizeMB} MB")

    // 如果分区数已经合理，跳过合并
    val optimalPartitions = math.max(1, df.rdd.getNumPartitions / 4)
    if (df.rdd.getNumPartitions <= optimalPartitions) {
      logger.info(s"分区数 ${df.rdd.getNumPartitions} 无需合并")
      return
    }

    // 计算新的分区数
    val targetPartitions = optimalPartitions

    // 重新分区并写入
    val repartitionedDf = df.repartition(targetPartitions)

    // 写入（覆盖模式）
    config.partitionColumn match {
      case Some(partitionCol) if targetDate != null =>
        repartitionedDf.write
          .format("parquet")
          .option("parquet.compression", "ZSTD")
          .option("parquet.compression.level", "6")
          .mode("overwrite")
          .partitionBy(partitionCol)
          .save(basePath)

      case _ =>
        repartitionedDf.write
          .format("parquet")
          .option("parquet.compression", "ZSTD")
          .option("parquet.compression.level", "6")
          .mode("overwrite")
          .save(basePath)
    }

    logger.info(s"合并完成: ${config.tableName}, 新分区数: $targetPartitions")
  }
}
```

### 5.2 小文件合并 Shell 脚本

```bash
#!/bin/bash
# small_file_merge.sh

SPARK_HOME=/opt/module/spark-3.1.3
JAR_PATH=/opt/module/taxi-etl/taxi-etl-4.0-SNAPSHOT-jar-with-dependencies.jar
TABLE_NAME=${1:-all}
TARGET_DATE=${2:-$(date -d "yesterday" +%Y-%m-%d)}

echo "开始小文件合并任务: TABLE=${TABLE_NAME}, DATE=${TARGET_DATE}"

${SPARK_HOME}/bin/spark-submit \
  --master yarn \
  --deploy-mode cluster \
  --class com.taxi.etl.common.SmallFileMergeTask \
  --driver-memory 4g \
  --executor-memory 4g \
  --executor-cores 2 \
  --num-executors 2 \
  ${JAR_PATH} \
  ${TABLE_NAME} \
  ${TARGET_DATE}

echo "小文件合并任务完成"
```
