
# Taxi Analytics Backend 模块分析报告

## 1. 模块概述

Taxi Analytics Backend 是纽约市出租车数据分析系统的**后端服务模块**，基于 Spring Boot 3.x 构建，提供 RESTful API、WebSocket 实时推送、AI 智能查询等核心功能。

### 1.1 核心功能

| 功能模块 | 描述 | 状态 |
|---------|------|------|
| **Dashboard** | 仪表盘 KPI 数据、趋势分析 | ✅ |
| **Analysis** | 多维度分析、供应商对比、机场统计等 | ✅ |
| **Realtime** | 实时指标、热点区域、费用构成、WebSocket 推送 | ✅ |
| **Map** | 热力图数据、区域映射 | ✅ |
| **Quality** | 数据质量报告、健康状态、告警 | ✅ |
| **AI** | 自然语言查询、SQL 生成、图表可视化、会话管理 | ✅ |
| **Export** | 数据导出任务管理 | ✅ |

### 1.2 技术栈

| 技术组件 | 版本 | 用途 |
|---------|------|------|
| Spring Boot | 3.2.x | Web 框架 |
| Spring WebSocket | - | 实时推送 |
| MyBatis Plus | - | ORM 框架 |
| MySQL | 8.0+ | 数据存储 |
| Swagger/OpenAPI | 3.0 | API 文档 |
| Coze/DeepSeek | - | AI 大模型 |

## 3. 核心实现分析

### 3.1 AI 智能查询系统

AI 模块是本系统的核心亮点，支持自然语言查询转 SQL 和图表可视化。

#### 3.1.1 处理流程

```java
@Service
public class AiServiceImpl implements AiService {
  
  @Override
  public AiQueryResponse processQuery(AiQueryRequest request) {
    long startTime = System.currentTimeMillis();
    
    // 1. 调用 LLM 获取响应
    LlmResponse llmResponse = llmService.callAgent(request.getQuery(), request.getSessionId());
    
    // 2. 智能路由
    if (llmResponse.isErrorResponse()) {
      return handleErrorResponse(request.getQuery(), llmResponse.getErrorMsg());
    } else if (llmResponse.isSqlResponse()) {
      return handleSqlResponse(request.getQuery(), llmResponse, startTime);
    } else {
      return handleDslResponse(request.getQuery(), llmResponse, startTime);
    }
  }
  
  // 处理 DSL 响应
  private AiQueryResponse handleDslResponse(String query, LlmResponse llmResponse, long startTime) {
    // a. DSL 验证
    Dsl dsl = dslConverter.convert(llmResponse.getDslString());
    dslValidator.validate(dsl);
    
    // b. 生成 SQL
    String sql = sqlGenerator.generateSqlFromDsl(dsl);
    
    // c. SQL 安全检查
    sqlSecurityGuard.validate(sql);
    
    // d. 执行查询
    List&lt;Map&lt;String, Object&gt;&gt; data = queryExecutor.execute(sql);
    
    // e. 生成图表配置
    ChartConfig chartConfig = sqlGenerator.suggestChart(dsl, data);
    
    // f. 构建响应
    return AiQueryResponse.builder()
      .query(query)
      .sql(sql)
      .data(data)
      .chartConfig(chartConfig)
      .executionTime(System.currentTimeMillis() - startTime)
      .hasChart(chartConfig != null)
      .build();
  }
}
```

```java
public class CozeClient implements LLMClient {
  
  @Override
  public String callAI(String prompt) {
    // 构建 SSE 请求
    HttpHeaders headers = new HttpHeaders();
    headers.setContentType(MediaType.APPLICATION_JSON);
    headers.set("Authorization", "Bearer " + apiKey);
    
    Map&lt;String, Object&gt; requestBody = new HashMap&lt;&gt;();
    requestBody.put("bot_id", botId);
    requestBody.put("user", "user-001");
    requestBody.put("query", prompt);
    requestBody.put("stream", true);
    
    // 解析 SSE 响应
    return parseSseResponse(restTemplate.postForObject(url, requestBody, String.class));
  }
  
  private String parseSseResponse(String sseResponse) {
    String[] lines = sseResponse.split("\n");
    for (String line : lines) {
      if (line.startsWith("data:")) {
        String data = line.substring(5).trim();
        // 解析 JSON，提取 content.text
        JsonNode json = objectMapper.readTree(data);
        if (json.has("content")) {
          return json.get("content").get("text").asText();
        }
      }
    }
    return null;
  }
}
```

#### 3.1.3 SQL 生成器

根据 DSL 生成 SQL：

```java
@Service
public class SqlGenerator {
  
  public String generateSqlFromDsl(Dsl dsl) {
    // 1. 确定目标表
    String tableName = determineTableName(dsl.getDimension(), dsl.getMetric());
    
    // 2. 获取表结构
    TableInfo tableInfo = schemaRetriever.getTableSchema(tableName);
    
    // 3. 解析指标和维度字段
    String metricField = resolveMetricField(tableInfo, dsl.getMetric());
    String dimensionField = resolveDimensionField(tableInfo, dsl.getDimension());
    
    // 4. 构建 SQL
    StringBuilder sql = new StringBuilder();
    sql.append("SELECT ").append(dimensionField).append(", ")
       .append("SUM(").append(metricField).append(") as value ");
    sql.append("FROM ").append(tableName).append(" ");
    sql.append("WHERE ").append(buildTimeCondition(dsl.getTimeRange()));
    sql.append("GROUP BY ").append(dimensionField).append(" ");
    sql.append("ORDER BY ").append(getOrderByClause(dimensionField));
    
    return sql.toString();
  }
  
  public ChartConfig suggestChart(Dsl dsl, List&lt;Map&lt;String, Object&gt;&gt; data) {
    String dimension = dsl.getDimension();
    
    // 根据维度类型推荐图表
    if (dimension.contains("date") || dimension.contains("hour")) {
      return ChartConfig.builder()
        .chartType("line")
        .xAxisField(dimension)
        .yAxisField("value")
        .title(dsl.getMetric() + " - " + dimension)
        .build();
    } else {
      return ChartConfig.builder()
        .chartType("bar")
        .xAxisField(dimension)
        .yAxisField("value")
        .title(dsl.getMetric() + " - " + dimension)
        .build();
    }
  }
}
```

```java
public class Result&lt;T&gt; {
  private Integer code;
  private String message;
  private T data;
  private Long timestamp;

    Result&lt;T&gt; result = new Result&lt;&gt;();
    result.setCode(ResultCode.SUCCESS.getCode());
    result.setMessage(ResultCode.SUCCESS.getMessage());
    result.setData(data);
    result.setTimestamp(System.currentTimeMillis());
    return result;
  }
  
  public static &lt;T&gt; Result&lt;T&gt; error(String message) {
    Result&lt;T&gt; result = new Result&lt;&gt;();
    result.setCode(ResultCode.ERROR.getCode());
    result.setMessage(message);
    result.setTimestamp(System.currentTimeMillis());
    return result;
  }
}

@RestControllerAdvice
public class GlobalExceptionHandler {
  
  @ExceptionHandler(BusinessException.class)
  public Result&lt;?&gt; handleBusinessException(BusinessException e) {
    log.error("业务异常: {}", e.getMessage());

  }
  
  @ExceptionHandler(Exception.class)
  public Result&lt;?&gt; handleException(Exception e) {
    log.error("系统异常: ", e);
    return Result.error("系统异常，请联系管理员");
  }
}
```

```sql
  order_count BIGINT,
  total_fare DECIMAL(18,2),
  avg_fare DECIMAL(18,2),
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  INDEX idx_window (window_start, window_end)
);
```
